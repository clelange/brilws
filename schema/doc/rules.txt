conventions:
  datatagnameid = 0; datatagname="online";
  
create tagname:
  insert into DATATAGS(datatagnameid,datatagname,creationutc,comments) values(datatagnameid,datatagname,creationutc,comments);

create tagid under a tagname:

  select datatagid from IDS_DATATAG where runnum=:runnum AND lsnum=:lsnum AND datatagnameid=:datatagnameid

  insert into IDS_DATATAG(datatagnameid,datatagid,fillnum,runnum,lsnum) values(datatagnameid,datatagid,fillnum,runnum,lsnum)
  
patchdata:
  create new tagname and register it
  create new tagid for the run+ls and register it under the new tagname
  insert new data under the new tagid into the specific table
